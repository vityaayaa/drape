Перенести логику из `sampleWallColors` — она не использует DOM кроме `document.createElement('canvas')` (заменить на `new OffscreenCanvas`).

**2. Обновить `src/utils/pixelizerSampler.js`:**

## Блок 2 — JSON экспорт/импорт проекта

### Проблема

Пользователь не может сохранить проект и передать его другому устройству.
`src/utils/projectIO.js` не создан (только упомянут в TEHC_STACK.md).

### Что сохранять/восстанавливать

Из `projectStore.js` прочитай что такое `getSnapshot()` и `restoreSnapshot(state)`.

Полный проект включает:
- Настройки room (стены, плитка, углы, маски) — в store
- tileColors — в store  
- photoSettings — в store
- **Фотографии** — в IndexedDB как Blob (нужно конвертировать в base64)

### Создать `src/utils/projectIO.js`

```js
import { loadPhoto, savePhoto } from '../store/persistence.js'
import { useProjectStore } from '../store/projectStore.js'

export async function exportProject() {
  const snapshot = useProjectStore.getState().getSnapshot()
  const photoIds = [...new Set(
    Object.values(snapshot.pixelizer?.photoSettings ?? {})
      .map(s => s?.photoId).filter(Boolean)
  )]

  const photos = {}
  for (const photoId of photoIds) {
    const blob = await loadPhoto(photoId)
    if (blob) {
      const ab = await blob.arrayBuffer()
      const b64 = btoa(String.fromCharCode(...new Uint8Array(ab)))
      photos[photoId] = { data: b64, type: blob.type }
    }
  }

  const payload = { version: 1, snapshot, photos }
  const json = JSON.stringify(payload)
  const blob = new Blob([json], { type: 'application/json' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = `drape-project-${Date.now()}.json`
  document.body.appendChild(a)
  a.click()
  document.body.removeChild(a)
  URL.revokeObjectURL(url)
}

export async function importProject(file) {
  const text = await file.text()
  let payload
  try {
    payload = JSON.parse(text)
  } catch {
    throw new Error('Невалидный JSON')
  }

  if (payload.version !== 1 || !payload.snapshot) {
    throw new Error('Неверный формат файла')
  }

  // Минимальная валидация
  const { snapshot } = payload
  if (!Array.isArray(snapshot.walls) || typeof snapshot.tile !== 'object') {
    throw new Error('Повреждённый файл проекта')
  }

  // Восстановить фото
  const photos = payload.photos ?? {}
  for (const [photoId, { data, type }] of Object.entries(photos)) {
    const bytes = Uint8Array.from(atob(data), c => c.charCodeAt(0))
    const blob = new Blob([bytes], { type })
    await savePhoto(photoId, blob)
  }

  useProjectStore.getState().restoreSnapshot(snapshot)
}
```

import { describe, it, expect } from 'vitest'
import { buildSchemaSVG } from './schemaSVGBuilder.js'

const makeTile = () => ({ tile_width: '20', tile_height: '20', grout_width: '2', grout_color: '#ccc' })
const makeWall = (id, length, height) => ({
  id, name: `Стена ${id}`, length, height, masks: [], tile_overrides: {}

describe('buildSchemaSVG', () => {
  it('возвращает строку SVG', () => {
    const walls = [makeWall('w1', '300', '250')]
    const palette = [{ index: 1, hex: '#ff0000', count: 10, byWall: [] }]

    const result = buildSchemaSVG({ walls, tile: makeTile(), tileColors, palette })
    expect(result).toMatch(/^<svg/)
    expect(result).toMatch(/<\/svg>$/)
  })

  it('количество rect-элементов соответствует колонкам × рядам', () => {
    const wall = makeWall('w1', '44', '22')  // 44мм × 22мм
    const tile = { tile_width: '20', tile_height: '20', grout_width: '2', grout_color: '#ccc' }
    // columns = ceil((440+2)/(20+2)) = ceil(442/22) = 21... нет: 44мм стена
    // cols = ceil((440+2)/(20+2)) = ceil(442/22) = ceil(20.09) = 21... стена в мм = 440мм
    // Используем маленькую стену: 22мм стена, 10мм плитка, 0 шов
    // ...вычисли и проверь что SVG содержит правильное кол-во <rect>
    const palette = []
    const tileColors = {}
    const result = buildSchemaSVG({ walls: [wall], tile, tileColors, palette })
    expect(result).toContain('<rect')
  })

  it('режим real: содержит width="...mm"', () => {
    const walls = [makeWall('w1', '300', '250')]
    const result = buildSchemaSVG({ walls, tile: makeTile(), tileColors: {}, palette: [], options: { scale: 'real' } })
    expect(result).toMatch(/width="\d+mm"/)
  })

  it('режим fit: width="100%"', () => {
    const walls = [makeWall('w1', '300', '250')]
    const result = buildSchemaSVG({ walls, tile: makeTile(), tileColors: {}, palette: [], options: { scale: 'fit' } })
    expect(result).toContain('width="100%"')
  })
})
